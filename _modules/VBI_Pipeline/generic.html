    [docs]def average_data(data, axis=None, mask=None):
    """A very basic function to average ndarrays.
    This is basically a wrapper for np.mean() that is also mask aware and forces 32 bit floats.
    """
    if axis is None:
        axis = len(data.shape) - 1
    if mask is None:
        return np.mean(data, axis=axis, dtype=np.float32)
    else:
        return np.mean(data * mask, axis=axis, dtype=np.float32)
    



    [docs]def average_FitsData(fitsData, preserve_camera_pos=True):
    """Average a Data.FitsData object

    In the most general case this will simply collapse along the last axis of the stacked data (i.e., collapse all
    hdus). However, there is also the option to preserve VBI camera positions, in which case each camera position is
    collapsed separately.

    In the first case the header information is taken from the first data hdu, and in the second case the header for
    each camera position is taken from the first data hdu at that position.

    Parameters
    ----------
    fitsData : Data.FitsData
        Should probably actually have some data in it

    preserve_camera_pos : bool
        If True then keep the separate camera positions separate during the average

    Returns
    -------
    Data.FitsData
        The averaged data
    """
    numx, numy = fitsData.data[0].shape
    hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]

    if preserve_camera_pos:
        for i in fitsData.unique_step_list:
            output = np.zeros((numx, numy), dtype=np.float32)
            pos_idx = fitsData.find_data_keyword_idx(S122['CurrentSpatialStep'], i)
            if len(pos_idx) > 0:
                header = fitsData.data_hdus[pos_idx[0]].header
                print('{}: Averaging camera position {}'.format(tag(), i))
                for j in pos_idx:
                    print('{}:  using data HDU {}'.format(tag(), j))
                    output += fitsData.data[j]
                hdulist.append(pyfits.ImageHDU(output / len(pos_idx), header))

    else:
        print('{}: Averaging all HDUs'.format(tag()))
        output = np.zeros((numx, numy), dtype=np.float32)
        for d in fitsData.data:
            output += d
        hdulist.append(pyfits.ImageHDU(output / len(fitsData.data), fitsData.headers[1]))

    newData = Data.FitsData()
    newData.hdulist = hdulist
    newData.origin = 'averageFitsData'

    return newData
    



    [docs]def dark_correction(fitsData, dark_calibration):
    """Subtract a dark calibration from input data

    Dark subtraction is done on a per-camera-position basis and the result is forced to be 32 bit float.

    Parameters
    ----------
    fitsData : Data.FitsData
        The data the dark corrected

    dark_calibration : Data.FitsData
        A processed DarkCalibration

    Returns
    -------
    None
        The input fitsData is updated inplace.
    """
    # Should probably have a check here to make sure there are the same number of camera positions in both inputs
    print('{}: performing dark correction on {} with {}'.format(tag(), fitsData, dark_calibration))

    for pos in fitsData.unique_step_list:
        try:
            dark_data_idx = dark_calibration.find_data_keyword_idx(S122['CurrentSpatialStep'], pos)[0]
        except IndexError:
            raise IndexError('Could not find a Dark HDU with {} = {}'.format(S122['CurrentSpatialStep'], pos))
        print('{}: using dark data HDU {}:'.format(tag(), dark_data_idx))
        dark_data = dark_calibration.data[dark_data_idx]

        data_idx = fitsData.find_data_keyword_idx(S122['CurrentSpatialStep'], pos)
        for i in data_idx:
            print('{}:  using data HDU {}'.format(tag(), i))
            fitsData.data[i] = fitsData.data[i] - dark_data
    



    [docs]def gain_correction(fitsData, gain_calibration):
    """
    Divide a gain calibration into input data

    Gain correction is done on a per-camera-position basis and any NaN's in the result are set to 0. Note that dividing
    by 0 is allowed and will result in infinity.

    Parameters
    ----------
    fitsData : Data.FitsData
        The data to be gain corrected

    gain_calibration : Data.FitsData
        Data containing a processed GainCalibration

    Returns
    -------
    None
        The input fitsData is updated inplace.
    """
    print('{}: performing gain correction on {} with {}'.format(tag(), fitsData, gain_calibration))

    for pos in fitsData.unique_step_list:
        try:
            gain_data_idx = gain_calibration.find_data_keyword_idx(S122['CurrentSpatialStep'], pos)[0]
        except IndexError:
            raise IndexError('Could not find a Gain HDU with {} = {}'.format(S122['CurrentSpatialStep'], pos))
        print('{}: using gain data HDU {}:'.format(tag(), gain_data_idx))
        gain_data = gain_calibration.data[gain_data_idx]

        data_idx = fitsData.find_data_keyword_idx(S122['CurrentSpatialStep'], pos)
        for i in data_idx:
            print('{}:  using data HDU {}'.format(tag(), i))
            fitsData.data[i] = fitsData.data[i] / gain_data
            fitsData.data[i][fitsData.data[i] != fitsData.data[i]] = 0.

    [docs]def bin_calibration(Data, x_binning, y_binning):
    if len(Data.data[0].shape) != 2:
        raise TypeError("Binning currently only supports arrays with exactly 2 dimensions")

    data = Data.data[0]
    numx = Data.headers[0]['numx']
    numy = Data.headers[0]['numy']
    x_binning_old = Data.headers[0]['x_bin']
    y_binning_old = Data.headers[0]['y_bin']

    if x_binning_old > x_binning or y_binning_old > y_binning:
        raise ValueError(
                "Cannot bin to larger sizes\n(Attempted {}x{} -> {}x{})".format(x_binning_old, y_binning_old,
                                                                                x_binning, y_binning))
    if x_binning / x_binning_old % 1 != 0 or y_binning / y_binning_old % 1 != 0 \
            or x_binning % 1 != 0 or y_binning % 1 != 0:
        raise ValueError("Cannot bin by non-integer amounts\n(Attempted {}x{})".format(x_binning / x_binning_old,
                                                                                       y_binning / y_binning_old))

    x_step = x_binning // x_binning_old
    y_step = y_binning // y_binning_old

    xbinned_data = np.zeros((numx // x_step, numy))
    binned_data = np.zeros((numx // x_step, numy // y_step))

    for i in range(x_step):
        xbinned_data += data[i::x_step, :]
    for j in range(y_step):
        binned_data += xbinned_data[:, j::y_step]

    Data.data[0] = binned_data
    Data.update_header('numx', numx // x_step, overwrite=True)
    Data.update_header('numy', numy // y_step, overwrite=True)
    Data.update_header('x_bin', x_binning, overwrite=True)
    Data.update_header('y_bin', y_binning, overwrite=True)

    return
    



    [docs]def extract_ROI(Data, ROI_min, ROI_max):
    if not ROI_min:
        ROI_min = [0, 0]
    if not ROI_max:
        ROI_max = [-1, -1]

    Data.update_header('numx', ROI_max[0] - ROI_min[0], overwrite=True)
    Data.update_header('numy', ROI_max[1] - ROI_min[1], overwrite=True)
    Data.update_header('ROI_min', ROI_min)
    Data.update_header('ROI_max', ROI_max)

    Data.data = Data.data[ROI_min[0]:ROI_max[0], ROI_min[1]: ROI_max[1]]

    return
