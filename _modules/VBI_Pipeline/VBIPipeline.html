    [docs]def main(config_file):
    """Check for file existence and generate intermediate Calibration data as needed, then run the Science Calibration.

    Parameters for running the pipeline are read from a provided INI-style configuration file. The necessary
    parameters and format can be found in gen_blank_config(), in this file.

    At a minimum, all of the options in the '[Main]' section need to be filled in. If any of the intermediate
    Calibration files are missing (i.e., dark, gain, target) then the pipeline will call the associated Calibration
    task in an attempt to generate them. It is possible to re-run these steps by simply deleting the associated file.

    Parameters
    ----------
    config_file : str
        The path to an INI-style configuration file

    Returns
    -------
    Int
        0 if successful, 1 otherwise

    """
    t1 = time.time()
    config = ConfigParser()
    config.read(config_file)
    VBI_Pipeline.FLUSH = config.getboolean('Main', 'flush')
    print('{}: Loaded configuration file {}'.format(tag(), config_file))
    print('{}: FLUSH set to {}'.format(tag(), VBI_Pipeline.FLUSH))

    # Check to make sure raw science exists
    if not os.path.exists(config['Main']['raw_sci_dir']):
        print('{}: Raw science data directory {} does not exist!'.format(tag(), config['Main']['raw_sci_dir']))
        return 1

    suffix = CONSTANTS['common']['suffix']

    inst_name = get_instrument_name(config['Main']['raw_sci_dir'], suffix)
    print('{}: Instrument: {}'.format(tag(), inst_name))
    print('{}: DataSet: {}'.format(tag(), config['Main']['data_set_ID']))

    # Check for necessary calibration data and try to generate them if needed
    sci_kwargs = generate_intermediate_objects(config, inst_name)

    sci_kwargs['final_average'] = config['Main'].getboolean('final_average', False)

    run_science(config, sci_kwargs, suffix)
    print('{}: All done in {} s'.format(tag(), time.time() - t1))

    return 0
    



    [docs]def get_instrument_name(data_dir, suffix):
    """Given a directory of VBI data, determine which camera was used to take the data.

    This function does NOT verify that all of the data are from the same camera (that happens elsewhere), it just
    returns the instrument (camera) of the first file read by the system.

    Parameters
    ----------
    data_dir : str
        Directory containing VBI data

    suffix : str
        Only look for files with this suffix

    Returns:
    str
        The value of INSTRUME for the first file loaded. Will be 'VBI-RED' or 'VBI-BLUE'
    """
    FD = Data.ObservationData(data_dir, suffix=suffix)

    fits_inst = FD.headers[0][S122['Instrument']]

    for k in INSTRUMENT_STRINGS.keys():
        if fits_inst in INSTRUMENT_STRINGS[k]:
            return k

    raise ValueError('The value of {} ("{}") is not a recognized instrument'.format(S122['Instrument'], fits_inst))
    


    [docs]def run_science(config, sci_kwargs, suffix):
    """Determine if multiple targets are going to be reduced and then start the ScienceCalibration

    The user can reduce all targets (data sets) present by setting data_set_ID = all. In this case all unique data
    sets will be discovered and reduced in sequence. Otherwise, just call a single ScienceCalibration run on the
    provided data set.

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options

    sci_kwargs : dict
        Dictionary containing all keyword arguments for call to ScienceCalibration.main()

    Returns
    -------
    None
    """
    unique_targets = get_all_obs_targets(config['Main']['raw_sci_dir'], suffix=suffix)
    if config['Main']['data_set_ID'].lower() == 'all':
        print('{}: Found targets {} in {}'.format(tag(), unique_targets, config['Main']['raw_sci_dir']))

        outprefix = generic.strip_fits(config['Main']['output'])
        for target in unique_targets:
            DSPSData = Data.DSPSData(config['Main']['raw_sci_dir'], target)
            print('{}: Running target {}'.format(tag(), target))
            print('{}: Target {} has {} DSPS repeats'.format(tag(), target, DSPSData.num_dsps))

            for dsps in DSPSData.dsps_dict.keys():
                print('{}: Running DSPS repeat {}'.format(tag(), dsps))
                output = generic.ensure_fits('{:}_{:}_{:03n}'.format(outprefix, target.replace(' ', ''), dsps))
                args = [DSPSData, dsps, output]

                print('{}: Calling ScienceCalibration.main()'.format(tag()))
                print('{}:  with args {}'.format(tag(), args))
                print('{}:  and kwargs {}'.format(tag(), sci_kwargs))
                ScienceCalibration.main(*args, **sci_kwargs)

    else:
        if config['Main']['data_set_ID'] not in unique_targets:
            raise ValueError('data_set_ID "{}" does not exist in {}'.format(config['Main']['data_set_ID'],
                                                                            config['Main']['raw_sci_dir']))

        DSPSData = Data.DSPSData(config['Main']['raw_sci_dir'], config['Main']['data_set_ID'])
        print('{}: Target {} has {} DSPS repeats'.format(tag(), config['Main']['data_set_ID'], DSPSData.num_dsps))

        for dsps in DSPSData.dsps_dict.keys():
            print('{}: Running DSPS repeat {}'.format(tag(), dsps))
            output = generic.ensure_fits('{:}_{:03n}'.format(generic.strip_fits(config['Main']['output']), dsps))
            args = [DSPSData, dsps, output]

            print('{}: Running science data'.format(tag()))
            print('{}: Calling ScienceCalibration.main()'.format(tag()))
            print('{}:  with args {}'.format(tag(), args))
            print('{}:  and kwargs {}'.format(tag(), sci_kwargs))
            ScienceCalibration.main(*args, **sci_kwargs)

    return
    


    [docs]def get_all_obs_targets(data_dir, suffix='FITS'):
    """Find the unique set of data set ID's (ID___003) for Observation data in a given directory

    Only files with DKIST004 == 'Observation' will be scanned.

    Parameters
    ----------
    data_dir : str
        Path to directory containing data

    suffix : str
        Optional file mask. The full mask is DATA_DIR/*SUFFIX.

    Returns
    -------
    list
        A list containing all unique values of ID___003 found in the given directory
    """
    file_list = glob('{}/*{}'.format(data_dir, suffix))

    hdulist = [pyfits.open(f) for f in file_list]
    target_list = [h[0].header[S122['DataSetParametersID']] for h in hdulist if
                   h[0].header[S122['InstrumentProgramTask']] in TASK_DATA_DICTS['Observation']['InstrumentProgramTask']]
    unique_targets = np.unique(target_list)

    for h in hdulist:
        h.close()

    return list(unique_targets)
    


    [docs]def generate_intermediate_objects(config, instrument):
    """Generate/collect the intermediate Calibration Objects needed by the Science Calibration module.

    If the an individual object already exists on disk, simply read it. Otherwise, call that object's generating module.

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options

    instrument : str
        Name of the instrument. Use `get_instrument_name` to find this.

    Returns
    -------
    dict
        Dictionary containing all keyword arguments for a future ScienceCalibration call
    """
    # Dark
    ######
    filename = generic.ensure_fits(config['Main']['dark_cal'])
    if not os.path.exists(filename):
        print('{}: Could not find {}, attempting to generate with DarkCalibration'.format(tag(), filename))
        kwargs = get_intermediate_kwargs(config, instrument, 'DarkCalibration', 'raw_dark_dir')
        args = [config['DarkCalibration']['raw_dark_dir'], filename]
        print('{}: Calling DarkCalibration.main()'.format(tag()))
        print('{}:  with args {}'.format(tag(), args))
        print('{}:  and kwargs {}'.format(tag(), kwargs))
        DarkCal = DarkCalibration.main(*args, **kwargs)
    else:
        DarkCal = Data.FitsData(filename)
        print('{}: Loaded {}'.format(tag(), filename))

    # Gain
    ######
    filename = generic.ensure_fits(config['Main']['gain_cal'])
    if not os.path.exists(filename):
        print('{}: Could not find {}, attempting to generate with GainCalibration'.format(tag(), filename))
        args = [config['GainCalibration']['raw_gain_dir'], filename]
        kwargs = get_intermediate_kwargs(config, instrument, 'GainCalibration', 'raw_gain_dir')
        kwargs['dark_cal'] = DarkCal
        print('{}: Calling GainCalibration.main()'.format(tag()))
        print('{}:  with args {}'.format(tag(), args))
        print('{}:  and kwargs {}'.format(tag(), kwargs))
        GainCal = GainCalibration.main(*args, **kwargs)
    else:
        GainCal = Data.FitsData(filename)
        print('{}: Loaded {}'.format(tag(), filename))

    # Target
    ########
    if config['Main'].getboolean('apply_target_cal', True):
        filename = generic.ensure_fits(config['Main']['target_cal'])
        if not os.path.exists(filename):
            print('{}: Could not find {}, attempting to generate with TargetCalibration'.format(tag(), filename))
            args = [config['TargetCalibration']['raw_grid_dir'], filename]
            kwargs = get_intermediate_kwargs(config, instrument, 'TargetCalibration', 'raw_grid_dir')
            kwargs['dark_cal'] = DarkCal
            kwargs['gain_cal'] = GainCal
            print('{}: Calling TargetCalibration.main()'.format(tag()))
            print('{}:  with args {}'.format(tag(), args))
            print('{}:  and kwargs {}'.format(tag(), kwargs))
            TargetCal = TargetCalibration.main(*args, **kwargs)
        else:
            TargetCal = Data.FitsData(filename)
            print('{}: Loaded {}'.format(tag(), filename))
    else:
        TargetCal = None

    # Construct argument list for the main pipeline
    sci_kwargs = {'dark_cal': DarkCal, 'gain_cal': GainCal, 'target_cal': TargetCal}

    return sci_kwargs
    


    [docs]def get_intermediate_kwargs(config, instrument, section, data_kwd):
    """Given user-defined and default parameters, construct the kwarg dictionary for a call to a *Calibration module

    User-provided parameters take precedence over defaults.

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options

    instrument : str
        Name of the instrument. Directly the 'INSTRUME' header keyword.

    section : str
        Name of the section in the config object. This relates to the specific Calibration object being called.

    data_kwd : str
        Name of the config option that points to the actual data to be used. This keyword is excluded from the output
        dictionary because it will already be in the main argument list.

    Returns
    -------
    dict
        Dictionary containing the keyword arguments that have been loaded from config and defaults for the specified
        section
    """
    kwargs = {}
    def_sec_name = '{}.{}'.format(instrument, section)

    # The default file...
    if def_sec_name in CONSTANTS.keys():
        for kw in CONSTANTS[def_sec_name].keys():
            # literal_eval is a safe way of turning e.g. strings into float
            try:
                kwargs[kw] = ast.literal_eval(CONSTANTS[def_sec_name][kw])
            except ValueError:
                kwargs[kw] = CONSTANTS[def_sec_name][kw]

    # ...and allow the user to override with their own definitions
    for kw in config[section].keys():
        if kw != data_kwd:
            try:
                kwargs[kw] = ast.literal_eval(config[section][kw])
            except ValueError:
                kwargs[kw] = config[section][kw]

    return kwargs
    


    [docs]def gen_blank_config(filename):
    """Create a blank configuration file.

    This is useful as a way to see all of the available options.

    Parameters
    ----------
    filename : str
        The location to save the blank file. Existing files will be overwritten.

    Returns
    -------
    None
    """
    config = ConfigParser(allow_no_value=True)
    config['Main'] = {'raw_sci_dir': '',
                      'data_set_ID': '',
                      'output': '',
                      "; The following 3 options point to where these objects will be saved. They don't need to exist": None,
                      'dark_cal': '',
                      'gain_cal': '',
                      'target_cal': '',
                      'apply_target_cal': '',
                      'final_average': '',
                      '; if flush is True memory usage will be very conservative at the expense of more disk usage': None,
                      'flush': False}
    config['DarkCalibration'] = {'raw_dark_dir': ''}
    config['GainCalibration'] = {'raw_gain_dir': ''}
    config['TargetCalibration'] = {'raw_grid_dir': ''}

    with open(filename, 'w') as f:
        config.write(f)

    return

def command_line():
    import argparse
    import sys

    parser = argparse.ArgumentParser(
        description='Run the VBI Pipeline using the parameters specified in the supplied config file. The user MUST '
                    'specify either a set of options or a valid config file. Config files follow the standard INI '
                    'format.')
    parser.add_argument('config_file', help="INI-style file containing pipeline options. Default is 'config.ini'.",
                        nargs='?', default='config.ini')
    gengroup = parser.add_mutually_exclusive_group()
    gengroup.add_argument('-b', '--blank', action='store_true', help='Generate a blank configuration file')
    gengroup.add_argument('-d', '--default', action='store_true',
                          help='Generate a configuration file with a some sensible parameters')
    args = parser.parse_args()

    if args.default:
        gen_default_config(args.config_file)
        sys.exit(0)

    if args.blank:
        gen_blank_config(args.config_file)
        sys.exit(0)

    if os.path.exists(args.config_file):
        sys.exit(main(args.config_file))
    else:
        print('Could not load config file {}. Aborting'.format(args.config_file))
        sys.exit(1)
